define([
	"../navigator",
	"poly/array"
], function SequenceModule(Navigator) {
	"use strict";

	/**
	 * @class browser.route.runner.sequence
	 * @extends core.event.emitter.runner
	 * @protected
	 * @static
	 */

	var UNDEFINED;
	var CONTEXT = "context";
	var CALLBACK = "callback";
	var TYPE = "type";
	var DATA = "data";
	var HEAD = "head";
	var NEXT = "next";
	var NAVIGATOR = "navigator";
	var MODIFIED = "modified";

	/**
	 * @method constructor
	 * @static
	 * @inheritdoc
	 * @localdoc Runner that executes ROUTE candidates in sequence without overlap
	 * @returns {*} Result from last handler
	 */
	return function sequence(event, handlers, args) {
		var modified = handlers[MODIFIED];
		var path = args[0];
		var navigator;
		var candidate;

		// Try get NAVIGATOR from handlers and check if MODIFIED
		if ((navigator = handlers[NAVIGATOR]) === UNDEFINED || navigator[MODIFIED] !== modified) {
			// Create and cache NAVIGATOR
			navigator = handlers[NAVIGATOR] = Navigator();

			// Set MODIFIED on navigator
			navigator[MODIFIED] = modified;

			// Iterate handlers
			for (candidate = handlers[HEAD]; candidate !== UNDEFINED; candidate = candidate[NEXT]) {
				// Add candidate to navigator
				navigator.add(candidate[TYPE], candidate);
			}
		}

		return navigator
			// Filter to only routes that match path
			.matches(path)
			// Reduce so we can catch the end value
			.reduce(function (result, navigator) {

				// Get candidate from navigator
				var candidate = navigator[1];

				// Run candidate, provide result to next run
				return candidate[CALLBACK].apply(candidate[CONTEXT], args);
			}, UNDEFINED);
	}
});
